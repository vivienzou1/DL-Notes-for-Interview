def bar(s):
    while "%" in s:
        i = s.find("%")
        j = i - 1
        while s[j].isdigit():
            j -= 1
        
        s = s[:j+1] + str(int(s[j+1: i]) / 100) + s[i+1:]
    return s
    
def foo(s):
    if "**" in s or "++" in s or "--" in s:
        return "error"
    
    s = bar(s)
    
    try:
        ans = eval(s)
        if isinstance(ans, int):
            return "%d" % ans
        else:
            return "%0.3f" % ans
